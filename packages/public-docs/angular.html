Angular Installation

Add Stencil components to your Angular application

Installation

Terminal 
npm install @your-org/angular-library

Module Setup

Add CUSTOM_ELEMENTS_SCHEMA to your app module:

app.module.ts 
import { NgModule, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core';
import { BrowserModule } from '@angular/platform-browser';
import { MyComponent } from '@your-org/angular-library';
import { AppComponent } from './app.component';

@NgModule({
  declarations: [AppComponent],
  imports: [BrowserModule, MyComponent],
  schemas: [CUSTOM_ELEMENTS_SCHEMA],
  bootstrap: [AppComponent]
})
export class AppModule {}

Standalone Components

For standalone Angular components:

app.component.ts 
import { Component, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core';
import { MyComponent } from '@your-org/angular-library';

@Component({
  selector: 'app-root',
  standalone: true,
  imports: [MyComponent],
  schemas: [CUSTOM_ELEMENTS_SCHEMA],
  templateUrl: './app.component.html'
})
export class AppComponent {}

Basic Usage

app.component.html 
<my-button variant="primary" size="large">Click Me</my-button>

<my-card cardTitle="Welcome" subtitle="Get started" elevation="high">
  <p>Card content goes here</p>
  <div slot="footer">
    <my-badge color="success">New</my-badge>
  </div>
</my-card>

Event Binding

Component 
import { Component } from '@angular/core';

@Component({
  selector: 'app-root',
  template: `
    <div>
      <p>Count: {{ count }}</p>
      <my-button 
        variant="primary" 
        (click)="increment()"
      >
        Increment
      </my-button>
    </div>
  `
})
export class AppComponent {
  count = 0;

  increment() {
    this.count++;
  }
}

Property Binding

Property Binding 
import { Component } from '@angular/core';

@Component({
  selector: 'app-root',
  template: `
    <my-button 
      [variant]="buttonVariant" 
      [disabled]="isDisabled"
      (click)="handleClick()"
    >
      {{ buttonText }}
    </my-button>
  `
})
export class AppComponent {
  buttonVariant: 'primary' | 'secondary' | 'danger' = 'primary';
  isDisabled = false;
  buttonText = 'Click Me';

  handleClick() {
    console.log('Button clicked!');
  }
}

Complete Example

Full Example 
import { Component } from '@angular/core';

@Component({
  selector: 'app-dashboard',
  template: `
    <div style="padding: 20px">
      <h1>Dashboard</h1>
      
      <my-card 
        cardTitle="Notifications" 
        [subtitle]="'You have ' + notifications + ' new notifications'"
        elevation="medium"
      >
        <p>Check your latest updates</p>
        
        <div slot="footer">
          <my-button 
            variant="primary" 
            size="small"
            (click)="markAllRead()"
          >
            Mark All Read
          </my-button>
          <my-badge color="warning">{{ notifications }}</my-badge>
        </div>
      </my-card>
    </div>
  `
})
export class DashboardComponent {
  notifications = 3;

  markAllRead() {
    this.notifications = 0;
  }
}
