React Installation

Add Stencil components to your React application

Installation

Terminal 
npm install @stencil-kit/react

Basic Usage

App.tsx 
import { MyButton, MyCard, MyBadge } from '@stencil-kit/react';

function App() {
  return (
    <div>
      <MyButton variant="primary" size="large">
        Click Me
      </MyButton>
      
      <MyCard cardTitle="Welcome" subtitle="Get started" elevation="high">
        <p>Card content goes here</p>
        <div slot="footer">
          <MyBadge color="success">New</MyBadge>
        </div>
      </MyCard>
    </div>
  );
}

export default App;

With TypeScript

Full TypeScript support is included out of the box:

TypeScript 
import { MyButton } from '@stencil-kit/react';
import type { JSX } from '@stencil-kit/react';

// Props are fully typed with autocomplete
const MyComponent: React.FC = () => {
  const handleClick = () => {
    console.log('Button clicked!');
  };

  return (
    <MyButton 
      variant="primary"  // ✅ Autocomplete works!
      size="medium"      // ✅ Type-safe
      onClick={handleClick}
    >
      Click Me
    </MyButton>
  );
};

Event Handling

Events 
import { MyButton } from '@stencil-kit/react';
import { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Count: {count}</p>
      <MyButton 
        variant="primary" 
        onClick={() => setCount(count + 1)}
      >
        Increment
      </MyButton>
    </div>
  );
}

Complete Example

Full Example 
import { MyButton, MyCard, MyBadge } from '@stencil-kit/react';
import { useState } from 'react';

function Dashboard() {
  const [notifications, setNotifications] = useState(3);

  return (
    <div style={{ padding: '20px' }}>
      <h1>Dashboard</h1>
      
      <MyCard 
        cardTitle="Notifications" 
        subtitle={`You have ${notifications} new notifications`}
        elevation="medium"
      >
        <p>Check your latest updates</p>
        
        <div slot="footer">
          <MyButton 
            variant="primary" 
            size="small"
            onClick={() => setNotifications(0)}
          >
            Mark All Read
          </MyButton>
          <MyBadge color="warning">{notifications}</MyBadge>
        </div>
      </MyCard>
    </div>
  );
}

export default Dashboard;
