Vanilla JavaScript Installation

The simplest way to use Stencil components

CDN Installation (Recommended)

No build step required! Just add a script tag:

index.html 
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>My App</title>
</head>
<body>
  <h1>My Application</h1>
  
  <!-- Use components -->
  <my-button variant="primary" size="medium">Click Me</my-button>
  
  <my-card card-title="Welcome" subtitle="Get started" elevation="medium">
    <p>This is a card component</p>
    <div slot="footer">
      <my-badge color="success">New</my-badge>
    </div>
  </my-card>

  <!-- Load from CDN -->
  <script type="module" src="https://unpkg.com/@stencil-kit/core/dist/stencil-library/stencil-library.esm.js"></script>
</body>
</html>

NPM Installation

For projects with a build step:

Terminal 
npm install @stencil-kit/core

Then import in your JavaScript:

main.js 
import { defineCustomElements } from '@stencil-kit/core/loader';

// Register components
defineCustomElements();

JavaScript Interaction

JavaScript 
<script type="module">
  // Wait for components to be defined
  await customElements.whenDefined('my-button');
  
  // Get reference to component
  const button = document.querySelector('my-button');
  
  // Add event listener
  button.addEventListener('click', () => {
    console.log('Button clicked!');
  });
  
  // Change props dynamically
  button.variant = 'danger';
  button.size = 'large';
</script>

Dynamic Creation

Create Elements 
// Create button element
const button = document.createElement('my-button');
button.variant = 'primary';
button.size = 'large';
button.textContent = 'Click Me';

// Add event listener
button.addEventListener('click', () => {
  alert('Button clicked!');
});

// Append to DOM
document.body.appendChild(button);

Complete Example

Full HTML Page 
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Stencil Components Demo</title>
  <style>
    body {
      font-family: Arial, sans-serif;
      max-width: 1200px;
      margin: 0 auto;
      padding: 40px;
      background: #f5f5f5;
    }
    .demo-section {
      margin: 40px 0;
    }
  </style>
</head>
<body>
  <h1>Stencil Design System Demo</h1>
  
  <div class="demo-section">
    <h2>Buttons</h2>
    <my-button variant="primary" size="small">Small</my-button>
    <my-button variant="primary" size="medium">Medium</my-button>
    <my-button variant="primary" size="large">Large</my-button>
  </div>

  <div class="demo-section">
    <h2>Badges</h2>
    <my-badge color="success">Success</my-badge>
    <my-badge color="warning">Warning</my-badge>
    <my-badge color="error">Error</my-badge>
  </div>

  <div class="demo-section">
    <h2>Interactive Card</h2>
    <my-card id="notification-card" card-title="Notifications" subtitle="You have 3 new messages" elevation="high">
      <p>Check your latest updates.</p>
      <div slot="footer">
        <my-button id="mark-read-btn" variant="primary" size="small">Mark All Read</my-button>
        <my-badge id="count-badge" color="warning">3</my-badge>
      </div>
    </my-card>
  </div>

  <script type="module" src="https://unpkg.com/@stencil-kit/core/dist/stencil-library/stencil-library.esm.js"></script>
  
  <script type="module">
    // Wait for components to load
    await customElements.whenDefined('my-button');
    
    // Add interactivity
    const markReadBtn = document.getElementById('mark-read-btn');
    const countBadge = document.getElementById('count-badge');
    const card = document.getElementById('notification-card');
    
    markReadBtn.addEventListener('click', () => {
      countBadge.textContent = '0';
      card.subtitle = 'All caught up!';
    });
  </script>
</body>
</html>

Browser Support

Stencil components work in all modern browsers:

  • ✅ Chrome/Edge (latest)
  • ✅ Firefox (latest)
  • ✅ Safari (latest)
  • ✅ Mobile browsers (iOS Safari, Chrome Mobile)

For older browsers, polyfills are automatically loaded when needed.
