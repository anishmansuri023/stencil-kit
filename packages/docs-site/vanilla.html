Vanilla HTML/JS Integration

The simplest way to use Stencil components

Overview

No build step required! Just load the script and use the components directly in your HTML.

Basic Usage

index.html 
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>My App</title>
</head>
<body>
  <h1>My Application</h1>
  
  <!-- Use components -->
  <my-button variant="primary" size="medium">Click Me</my-button>
  
  <my-card card-title="Welcome" subtitle="Get started" elevation="medium">
    <p>This is a card component</p>
    <div slot="footer">
      <my-badge color="success">New</my-badge>
    </div>
  </my-card>

  <!-- Load Stencil library -->
  <script type="module" src="../stencil-library/dist/stencil-library/stencil-library.esm.js"></script>
</body>
</html>

JavaScript Interaction

JavaScript 
<script type="module">
  // Wait for components to be defined
  await customElements.whenDefined('my-button');
  
  // Get reference to component
  const button = document.querySelector('my-button');
  
  // Add event listener
  button.addEventListener('click', () => {
    console.log('Button clicked!');
  });
  
  // Change props dynamically
  button.variant = 'danger';
  button.size = 'large';
</script>

Complete Example

Complete HTML Page 
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Stencil Components Demo</title>
  <style>
    body {
      font-family: Arial, sans-serif;
      max-width: 1200px;
      margin: 0 auto;
      padding: 40px;
      background: #f5f5f5;
    }
    .demo-section {
      margin: 40px 0;
    }
  </style>
</head>
<body>
  <h1>Stencil Design System Demo</h1>
  
  <div class="demo-section">
    <h2>Buttons</h2>
    <my-button variant="primary" size="small">Small</my-button>
    <my-button variant="primary" size="medium">Medium</my-button>
    <my-button variant="primary" size="large">Large</my-button>
  </div>

  <div class="demo-section">
    <h2>Badges</h2>
    <my-badge color="success">Success</my-badge>
    <my-badge color="warning">Warning</my-badge>
    <my-badge color="error">Error</my-badge>
  </div>

  <div class="demo-section">
    <h2>Card</h2>
    <my-card card-title="Example Card" subtitle="With all features" elevation="high">
      <p>This is the card content area.</p>
      <div slot="footer">
        <my-button variant="primary" size="small">Action</my-button>
        <my-badge color="info">New</my-badge>
      </div>
    </my-card>
  </div>

  <script type="module" src="../stencil-library/dist/stencil-library/stencil-library.esm.js"></script>
</body>
</html>
